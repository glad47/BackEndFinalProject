#import <Foundation/Foundation>;

@interface ResponseResult : NSObject

//是否成功
@property (nonatomic, strong) BOOL success;
//编码
@property (nonatomic, assign) NSString code;
//描述信息
@property (nonatomic, assign) NSString message;
//结果
@property (nonatomic, strong) PageResult result;


@end



#import <Foundation/Foundation>;

@interface PageResult : NSObject

//当前页号
@property (nonatomic, strong) NSInteger pageNo;
//每页行数
@property (nonatomic, strong) NSInteger pageSize;
//总记录数
@property (nonatomic, strong) long total;
//总页数
@property (nonatomic, strong) long pageNum;
//动态的内容
@property (nonatomic, strong) Object data;


@end
